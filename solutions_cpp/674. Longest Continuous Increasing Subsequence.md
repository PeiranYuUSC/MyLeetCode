## One pointer
## O(n), O(1)
```c++
class Solution {
public:
    int findLengthOfLCIS(vector<int>& nums) {
        int count = 1;
        int cur = 0;
        int res = 0;
        if (nums.size() < 2) return nums.size();
        
        while (cur < nums.size() - 1) {
            if (nums[cur] < nums[cur + 1]) {
                count++;
            } else {
                count = 1;
            }
            cur++; 
            res = max(count, res);
        }
        
        return res;
    }
};
```
