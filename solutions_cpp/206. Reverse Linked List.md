## Recursion
## O(n), O(n)
```c++
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        // Base case
        if (head == nullptr || head->next == nullptr) {
            return head;
        }
        
        ListNode* newHead = reverseList(head->next);
        
        head->next->next = head;
        head->next = nullptr;
        
        return newHead;
    }
};
```
