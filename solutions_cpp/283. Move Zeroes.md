## Two for loops without swap()
## O(n), O(1)
```c++
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int slow = 0;
        int n = nums.size();
        
        for (int i = 0; i < n; ++i) {
            if (nums[i] != 0) {
                nums[slow++] = nums[i];
            }
        }
        
        for (int j = slow; j < n; ++j) {
            nums[j] = 0;
        }
        
    }
};
```

## One for loop with swap
## O(n), O(1)
```c++
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int slow = 0;
        int n = nums.size();
        
        for (int i = 0; i < n; ++i) {
            if (nums[i] != 0) {
                swap(nums[slow++], nums[i]);
            }
        }
        
    }
};
```
