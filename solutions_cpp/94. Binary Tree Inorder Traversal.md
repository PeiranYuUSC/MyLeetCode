## Recursive
## O(n), O(n)
```c++
class Solution {
public:
    vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        inorder(root);
        return res;
    }
    
    void inorder(TreeNode* root) {
        // Base case
        if (root == nullptr) {
            return;
        }
        
        inorder(root->left);
        res.push_back(root->val);
        inorder(root->right);
    }
};
```
