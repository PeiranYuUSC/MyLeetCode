## Recursive
## O(n), O(n)
```c++
class Solution {
public:
    vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        inorder(root);
        return res;
    }
    
    void inorder(TreeNode* root) {
        // Base case
        if (root == nullptr) {
            return;
        }
        
        inorder(root->left);
        res.push_back(root->val);
        inorder(root->right);
    }
};
```

## Iterative with a stack
## O(n), O(n)
```c++
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> stk;
        TreeNode* curr = root;
        
        while (curr != nullptr || !stk.empty()) {
            while (curr != nullptr) {
                stk.push(curr);
                curr = curr->left;
            }
            curr = stk.top();
            stk.pop();
            res.push_back(curr->val);
            curr = curr->right;
        }
        
        return res;
    }
};
```
