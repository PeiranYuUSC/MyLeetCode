## O(n), O(1)
```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int slow = 0, fast = 0;
        int n = nums.size();
        while (fast < n) {
            if (nums[slow] != nums[fast]) {
                nums[++slow] = nums[fast];
            }
            fast++;
        }
        return slow + 1;
    }
};
```

## Two pointers ([fast] != [slow - 1])
## O(n), O(1)
```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int slow = 1;
        int n = nums.size();
        
        for (int fast = 1; fast < n; ++fast) {
            if (nums[fast] != nums[slow - 1]) {
                nums[slow++] = nums[fast];
            }
        }
        return slow;
    }
};
```
