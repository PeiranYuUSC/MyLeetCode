## Bit manipulation: Power the least significant bit and move to the right position
## O(Most significant 1 of the space), O(1)
```c++
class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t res = 0;
        int power = 31;
        
        while (n != 0) {
            if (n & 1 == 1) {
                res = res + (1 << power);
            }
            n = n >> 1;
            power -= 1;
        }
        
        return res;
    }
};
```
