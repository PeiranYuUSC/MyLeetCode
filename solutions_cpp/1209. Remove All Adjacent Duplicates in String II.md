## Stack to store counts
## O(n^2 / k), O(n)
```c++
class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<int> counts;
        
        for (int i = 0; i < s.size(); ++i) {
            // Push 1 when meet a different value
            if (i == 0 || s[i] != s[i - 1]) {
                counts.push(1);
            // Increment top() when meet a same value as the last one
            } else if (s[i] == s[i - 1]) {
                counts.top() ++;
            }
            
            // Pop when counts == k
            if (counts.top() == k) {
                counts.pop();
                // Erase k values
                s.erase(i - k + 1, k);
                i = i - k;
            }
        }
        
        return s;
    }
};
```
