## BFS with one queue
## O(n), O(n)
```c++
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        q.push(root);
        
        while (q.empty() == false) {
            TreeNode* curL = q.front();
            q.pop();
            TreeNode* curR = q.front();
            q.pop();
            
            if (curL == nullptr && curR == nullptr) continue;
            if (curL == nullptr || curR == nullptr || curL->val != curR->val) return false;
            
            q.push(curL->left);
            q.push(curR->right);
            q.push(curL->right);
            q.push(curR->left);
        }
        
        return true;
    }
    
};
```
