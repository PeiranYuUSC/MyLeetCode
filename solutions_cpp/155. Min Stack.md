## Two stacks
## O(1) for all operations, O(2n)
```c++
class MinStack {
public:
    // stkVal     1 2 3 0 5
    // stkMin MAX 1 1 1 0 0
    stack<int> stkVal;
    stack<int> stkMin;
    
    MinStack() {
        stkMin.push(INT_MAX);
    }
    
    void push(int val) {
        if (val <= stkMin.top()) {
            stkMin.push(val);
        } else {
            stkMin.push(stkMin.top());
        }
        stkVal.push(val);
    }
    
    void pop() {
        stkVal.pop();
        stkMin.pop();
    }
    
    int top() {
        return stkVal.top();
    }
    
    int getMin() {
        return stkMin.top();
    }
};

/**
 * Your MinStack object will be instantiated and called as such:
 * MinStack* obj = new MinStack();
 * obj->push(val);
 * obj->pop();
 * int param_3 = obj->top();
 * int param_4 = obj->getMin();
 */
```
