## Min Heap
## O(nlogk), O(k)

```c++
class Solution {
public:
    
    int findKthLargest(vector<int>& nums, int k) {
        // Use a min heap to have the kth largest value at top
        priority_queue<int, vector<int>, greater<int>> pq;
        
        
        for (int i = 0; i < nums.size(); i++) {
            pq.push(nums[i]);
            
            // Make a min heap of k values
            if (pq.size() > k) {
                pq.pop();
            }
            
        }
        
        return pq.top();
    }
};
```
