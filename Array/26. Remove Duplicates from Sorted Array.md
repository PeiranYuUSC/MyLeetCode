## Double Pointers

```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int fast = 0, slow = 0;
        
        while(fast < nums.size()) {
            if(nums[slow] == nums[fast]) {
                fast ++;
            } else {
                slow ++;
                nums[slow] = nums[fast];
                fast ++;
            }
        }
        
        return slow+1;
    }
};
```
